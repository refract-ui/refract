      brand: 'purple'
    })
  });

  const { colors } = theme({
    colors: {
      white: '#fefefe',

    }
  });
  return <ColorPalette colors={colors} />;
}

## Theme Configuration

Colors can be configured by extending the global styles via function or with static values. In each example, we replace the `colors.primary` with a new color and extend the colors object with a new key-value pair.

### Using Function Values

<Canvas showCode>

``` js
import { theme } from '@refract-ui/core';

const { colors } = theme({
  colors: ({ defaults }) => ({
    ...defaults,
    white: '#fefefe',
    brand: 'purple'
  })
});
```
