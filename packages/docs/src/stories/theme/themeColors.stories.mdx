export const Template = (args) => {
  const { primary, secondary, success } = args;
  const { themeColors } = theme({
    themeColors: ({ defaults }) => ({
      ...defaults,
      primary,
      secondary,
      success
    })
  });
  return <ColorPalette colors={themeColors} />;
}

export const FunctionTemplate = () => {
  const { themeColors } = theme({
    themeColors: ({ defaults }) => ({
      ...defaults,
      white: '#fefefe',
      primary: 'purple'
    })
  });
  return <ColorPalette colors={themeColors} />;
}

export const StaticTemplate = () => {
  const { themeColors } = theme({
    themeColors: {
      white: '#fefefe',
      primary: 'purple'
    }
  });
  return <ColorPalette colors={themeColors} />;
}

## Default Props

``` js
import { theme } from '@refract-ui/core';
const { themeColors } = theme({});

assert(themeColors).isEqual({
  danger: '#DE7283',
  dark: '#575C64',
  info: '#17a2b8',
  light: 'rgb(255, 255, 255)',
  primary: '#3978EC',
  secondary: '#B0B9C9',
  success: '#6AD193',
  warning: '#EFC174'
});
```

## Theme Configuration

`themeColors` can be configured by extending the global styles via function or with static values. In each example, we replace the `themeColors.primary` with a new color and extend the `themeColors` object with a new key-value pair.
